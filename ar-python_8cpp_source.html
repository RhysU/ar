54 " Usage: M = arsel(data, submean, absrho, criterion, minorder, maxorder)\n"
55 "\n"
56 " Automatically fit autoregressive models to input signals.\n"
57 "\n"
58 " Use ar::burg_method and ar::best_model to fit an autoregressive process\n"
59 " for signals contained in the rows of matrix data. Sample means will\n"
60 " be subtracted whenever submean is true. Model orders minorder through\n"
61 " min(columns(data), maxorder) will be considered. A dictionary is\n"
62 " returned where each key either contains a result indexable by the\n"
63 " signal number (i.e. the row indices of input matrix data) or it contains\n"
64 " a single scalar applicable to all signals.\n"
65 "\n"
66 " The model order will be selected using the specified criterion.\n"
67 " Criteria are specified using the following abbreviations:\n"
68 " AIC - Akaike information criterion\n"
69 " AICC - asymptotically-corrected Akaike information criterion\n"
70 " BIC - consistent criterion BIC\n"
71 " CIC - combined information criterion\n"
72 " FIC - finite information criterion\n"
73 " FSIC - finite sample information criterion\n"
74 " GIC - generalized information criterion\n"
75 " MCC - minimally consistent criterion\n"
76 "\n"
77 " The number of samples in data (i.e. the number of rows) is returned\n"
78 " in key 'N'. The filter()-ready process parameters are returned\n"
79 " in key 'AR', the sample mean in 'mu', and the innovation variance\n"
80 " \\sigma^2_\\epsilon in 'sigma2eps'. The process output variance\n"
81 " \\sigma^2_\\x and process gain are returned in keys 'sigma2x' and\n"
82 " 'gain', respectively. Autocorrelations for lags zero through the\n"
83 " model order, inclusive, are returned in key 'autocor'. The raw\n"
84 " signals are made available for later use in field 'data'.\n"
85 "\n"
86 " Given the observed autocorrelation structure, a decorrelation time\n"
87 " 'T0' is computed by ar::decorrelation_time and used to estimate\n"
88 " the effective signal variance 'eff_var'. The number of effectively\n"
89 " independent samples is returned in 'eff_N'. These effective values are\n"
90 " combined to estimate the sampling error (i.e. the standard deviation\n"
91 " of the sample mean) as field 'mu_sigma'. The absolute value of the\n"
92 " autocorrelation function will be used in computing the decorrelation\n"
93 " times whenever absrho is true.\n"

125 extern "C" PyObject *ar_arsel(PyObject *self, PyObject *args, PyObject *kwargs)
126 {
127  PyObject *ret_args = NULL, *ret = NULL;
128 
129  // Sanity check that initar could build ar_ArselType
130  if (!PyType_Check(ar_ArselType)) {
131  PyErr_SetString(PyExc_RuntimeError,
132  "PyType_Check(ar_ArselType) failed");
133  return NULL;
134  }
135 
136  // Prepare argument storage with default values
137  PyObject *data_obj = NULL;
138  int submean = DEFAULT_SUBMEAN;
139  int absrho = DEFAULT_ABSRHO;
140  const char *criterion = DEFAULT_CRITERION;
141  std::size_t minorder = DEFAULT_MINORDER;
142  std::size_t maxorder = DEFAULT_MAXORDER;
143 
144  // Define keyword argument names
145  static char *kwlist[] = {
146  (char *)"data",
147  (char *)"submean",
148  (char *)"absrho",
149  (char *)"criterion",
150  (char *)"minorder",
151  (char *)"maxorder",
152  NULL
153  };
154 
155  // Parse input tuple and keywords with second and subsequent arguments optional
156  {
157  unsigned long ul_minorder = minorder, ul_maxorder = maxorder;
158  if (!PyArg_ParseTupleAndKeywords(args, kwargs, "O|iiskk", kwlist,
159  &data_obj, &submean, &absrho,
160  &criterion,
161  &ul_minorder, &ul_maxorder)) {
162  return NULL;
163  }
164  minorder = ul_minorder;
165  maxorder = ul_maxorder;
166  }
167 
168  // Lookup the desired model selection criterion
169  typedef ar::best_model_function<
170  ar::Burg,npy_intp,npy_intp,std::vector<double>
171  > best_model_function;
172  const best_model_function::type best_model
173  = best_model_function::lookup(std::string(criterion), submean);
174  if (!best_model) {
175  PyErr_SetString(PyExc_RuntimeError,
176  "Unknown model selection criterion provided to arsel.");
177  return NULL;
178  }
179 
180  // Incoming data may be noncontiguous but should otherwise be well-behaved
181  // On success, 'data' is returned so Py_DECREF is applied only on failure
182  PyObject *data = PyArray_FROMANY(data_obj, NPY_DOUBLE, 1, 2,
183  NPY_ALIGNED | NPY_ELEMENTSTRIDES | NPY_NOTSWAPPED);
184  if (!data) {
185  Py_DECREF(data);
186  return NULL;
187  }
188 
189  // Reshape any 1D ndarray into a 2D ndarray organized as a row vector.
190  // Permits the remainder of the routine to uniformly worry about 2D only.
191  if (1 == PyArray_NDIM((PyArrayObject *)data)) {
192  npy_intp dim[2] = { 1, PyArray_DIM((PyArrayObject *)data, 0) };
193  PyArray_Dims newshape = { dim, sizeof(dim)/sizeof(dim[0]) };
194  PyObject *olddata = data;
195  data = PyArray_Newshape((PyArrayObject *)olddata,
196  &newshape, NPY_ANYORDER);
197  Py_DECREF(olddata);
198  if (!data) {
199  Py_DECREF(data);
200  PyErr_SetString(PyExc_RuntimeError,
201  "Unable to reorganize data into matrix-like row vector.");
202  return NULL;
203  }
204  }
205 
206  // How many data points are there?
207  npy_intp M = PyArray_DIM((PyArrayObject *)data, 0);
208  npy_intp N = PyArray_DIM((PyArrayObject *)data, 1);
209 
210  // Prepare per-signal storage locations to return to caller
211  // TODO Ensure these invocations all worked as expected
212  PyObject *_AR = PyList_New(M);
213  PyObject *_autocor = PyList_New(M);
214  PyObject *_eff_N = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
215  PyObject *_eff_var = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
216  PyObject *_gain = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
217  PyObject *_mu = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
218  PyObject *_mu_sigma = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
219  PyObject *_sigma2eps = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
220  PyObject *_sigma2x = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
221  PyObject *_T0 = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
222 
223  // Prepare vectors to capture burg_method() output
224  std::vector<double> params, sigma2e, gain, autocor;
225  params .reserve(maxorder*(maxorder + 1)/2);
226  sigma2e.reserve(maxorder + 1);
227  gain .reserve(maxorder + 1);
228  autocor.reserve(maxorder + 1);
229 
230  // Prepare repeatedly-used working storage for burg_method()
231  std::vector<double> f, b, Ak, ac;
232 
233  // Process each signal in turn...
234  for (npy_intp i = 0; i < M; ++i)
235  {
236  // Use burg_method to estimate a hierarchy of AR models from input data
237  params .clear();
238  sigma2e.clear();
239  gain .clear();
240  autocor.clear();
241  ar::strided_adaptor<const double*> signal_begin(
242  (const double*) PyArray_GETPTR2((PyArrayObject *)data, i, 0),
243  PyArray_STRIDES((PyArrayObject *)data)[1] / sizeof(double));
244  ar::strided_adaptor<const double*> signal_end (
245  (const double*) PyArray_GETPTR2((PyArrayObject *)data, i, N),
246  PyArray_STRIDES((PyArrayObject *)data)[1] / sizeof(double));
247  ar::burg_method(signal_begin, signal_end,
248  *(double*)PyArray_GETPTR1((PyArrayObject *)_mu, i),
249  maxorder,
250  std::back_inserter(params),
251  std::back_inserter(sigma2e),
252  std::back_inserter(gain),
253  std::back_inserter(autocor),
254  submean, /* output hierarchy? */ true, f, b, Ak, ac);
255 
256  // Keep only best model per chosen criterion via function pointer
257  try {
258  best_model(N, minorder, params, sigma2e, gain, autocor);
259  }
260  catch (std::exception &e)
261  {
262  PyErr_SetString(PyExc_RuntimeError, e.what());
263  goto fail;
264  }
265 
266  // Compute decorrelation time from the estimated autocorrelation model
267  ar::predictor<double> p = ar::autocorrelation(
268  params.begin(), params.end(), gain[0], autocor.begin());
269  *(double*)PyArray_GETPTR1((PyArrayObject *)_T0, i)
270  = ar::decorrelation_time(N, p, absrho);
271 
272  // Filter()-ready process parameters in field 'AR' with leading one
273  PyObject *_ARi = PyList_New(params.size() + 1);
274  PyList_SET_ITEM(_AR, i, _ARi);
275  PyList_SET_ITEM(_ARi, 0, PyFloat_FromDouble(1));
276  for (std::size_t k = 0; k < params.size(); ++k) {
277  PyList_SET_ITEM(_ARi, k+1, PyFloat_FromDouble(params[k]));
278  }
279 
280  // Field 'sigma2eps'
281  *(double*)PyArray_GETPTR1((PyArrayObject *)_sigma2eps, i) = sigma2e[0];
282 
283  // Field 'gain'
284  *(double*)PyArray_GETPTR1((PyArrayObject *)_gain, i) = gain[0];
285 
286  // Field 'sigma2x'
287  *(double*)PyArray_GETPTR1((PyArrayObject *)_sigma2x, i) = gain[0]*sigma2e[0];
288 
289  // Field 'autocor'
290  PyObject *_autocori = PyList_New(autocor.size());
291  PyList_SET_ITEM(_autocor, i, _autocori);
292  for (std::size_t k = 0; k < autocor.size(); ++k) {
293  PyList_SET_ITEM(_autocori, k, PyFloat_FromDouble(autocor[k]));
294  }
295 
296  // Field 'eff_var'
297  // Unbiased effective variance expression from [Trenberth1984]
298  *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_var, i)
299  = (N*gain[0]*sigma2e[0]) / (N - *(double*)PyArray_GETPTR1((PyArrayObject *)_T0, i));
300 
301  // Field 'eff_N'
302  *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_N, i)
303  = N / *(double*)PyArray_GETPTR1((PyArrayObject *)_T0, i);
304 
305  // Field 'mu_sigma'
306  // Variance of the sample mean using effective quantities
307  *(double*)PyArray_GETPTR1((PyArrayObject *)_mu_sigma, i)
308  = std::sqrt( *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_var, i)
309  / *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_N, i));
310  }
311 
312  // Prepare build and return an ar_ArselType via tuple constructor
313  // See initar(...) method for the collections.namedtuple-based definition
314  ret_args = PyTuple_Pack(17, PyBool_FromLong(absrho),
315  _AR,
316  _autocor,
317  PyUnicode_FromString(criterion),
318  data,
319  _eff_N,
320  _eff_var,
321  _gain,
322  PyLong_FromSize_t(maxorder),
323  PyLong_FromSize_t(minorder),
324  _mu,
325  _mu_sigma,
326  PyLong_FromLong(N),
327  _sigma2eps,
328  _sigma2x,
329  PyBool_FromLong(submean),
330  _T0);
331  if (!ret_args) {
332  PyErr_SetString(PyExc_RuntimeError,
333  "Unable to prepare arguments used to build return value.");
334  goto fail;
335  }
336  ret = PyObject_CallObject((PyObject *)ar_ArselType, ret_args);
337  if (!ret) {
338  PyErr_SetString(PyExc_RuntimeError,
339  "Unable to construct return value.");
340  goto fail;
341  }
342 
343  return ret;
344 
345 fail:
346  Py_XDECREF(ret_args);
347  Py_XDECREF(ret);
348  Py_XDECREF(data);
349  Py_XDECREF(_AR);
350  Py_XDECREF(_autocor);
351  Py_XDECREF(_eff_N);
352  Py_XDECREF(_eff_var);
353  Py_XDECREF(_gain);
354  Py_XDECREF(_mu);
355  Py_XDECREF(_mu_sigma);
356  Py_XDECREF(_sigma2eps);
357  Py_XDECREF(_sigma2x);
358  Py_XDECREF(_T0);
359  return NULL;
360 }
