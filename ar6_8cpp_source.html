54 // Box-Muller logic adjusted from
55 // https://github.com/rflynn/c/blob/master/rand-normal-distribution.c
56 static double rand01() {
57  const double x = (double) random() / RAND_MAX;
58  const double y = (double) random() / RAND_MAX;
59  const double z = sqrt(-2 * log(x)) * cos(2 * M_PI * y);
60  return z;
61 }
62 
63 // Scale to obtain variance 0.1.
64 static double rand0point1() {
65  return sqrt(0.1) * rand01();
66 }

68 int main(int argc, char *argv[])
69 {
70  using namespace std;
71 
72  long t = 0;
73  long burn = 500;
74  long tfinal = 3000;
75 
76  {
77  option::Stats stats(usage, argc-(argc>0), argv+(argc>0));
78 
79  vector<option::Option> opts(stats.options_max + stats.buffer_max);
80 
81  option::Parser parse(usage, argc-(argc>0), argv+(argc>0),
82  &opts[0], &opts[stats.options_max]);
83 
84  if (parse.error() || opts[UNKNOWN]) {
85  for (option::Option* o = opts[UNKNOWN]; o; o = o->next()) {
86  cerr << "Unknown option: " << o->name << "\n";
87  }
88  return EXIT_FAILURE;
89  }
90 
91  if (opts[HELP]) {
92  printUsage(cout, usage);
93  return EXIT_SUCCESS;
94  }
95 
96  // SEED must come before any other randomly-generated option
97  if (opts[SEED]) {
98  srandom((unsigned) strtol(opts[SEED].last()->arg, NULL, 10));
99  } else {
100  struct timeval tv;
101  struct timezone tz;
102  gettimeofday(&tv, &tz);
103  srand((unsigned) tv.tv_usec);
104  }
105 
106  // Parse remaining options
107  if (opts[BURN ]) burn = strtol(opts[BURN ].last()->arg, NULL, 10);
108  if (opts[TFINAL]) tfinal = strtol(opts[TFINAL].last()->arg, NULL, 10);
109 
110  // Warn whenever burn >= tfinal
111  if (burn >= tfinal) {
112  cerr << "Warning: burn >= tfinal suppresses output ("
113  << burn << " >= " << tfinal << ") \n";
114  }
115  }
116 
117  // Initialize the AR process parameters
118  // Negated as convention differs from https://arxiv.org/abs/1802.01056v1
119  std::vector<double> params;
120  params.push_back(-1 * +3.1378);
121  params.push_back(-1 * -3.9789);
122  params.push_back(-1 * +2.6788);
123  params.push_back(-1 * -1.0401);
124  params.push_back(-1 * +0.2139);
125  params.push_back(-1 * -0.0133);
126  ar::predictor<double> p(params.begin(), params.end(), rand0point1);
127 
128  std::vector<double> initial;
129  for (size_t i = 0; i < params.size(); ++i) {
130  initial.push_back(rand0point1());
131  }
132  p.initial_conditions(initial.begin());
133 
134  // Discard 0 <= t < burn
135  for (; t < burn; ++t, ++p) {
136  // NOP
137  }
138 
139  // Output (t, x, y, z) during burn < t <= tfinal
140  cout.precision(numeric_limits<double>::digits10 + 2);
141  cout << showpos;
142  for (; t < tfinal; ++t, ++p) {
143  cout << t << '\t' << *p << '\n';
144  }
145 
146  return EXIT_SUCCESS;
147 }
