79  // Parse and process any command line arguments using optionparser.h
80  string criterion = "CIC";
81  bool subtract_mean = false;
82  size_t minorder = 0;
83  size_t maxorder = 512;
84  bool absolute_rho = true;
85  double window_T0 = 1.0;
86  {

88 
89  vector<option::Option> options(stats.options_max + stats.buffer_max);

91  option::Parser parse(usage, argc-(argc>0), argv+(argc>0),
92  &options[0], &options[stats.options_max]);
93 
94  if (parse.error() || options[UNKNOWN]) {
95  for (option::Option* o = options[UNKNOWN]; o; o = o->next()) {
96  cerr << "Unknown option: " << o->name << "\n";

98  return EXIT_FAILURE;
99  }
100 
101  if (options[HELP]) {
102  printUsage(cout, usage);
103  return EXIT_SUCCESS;
104  }
105 
106  if (options[VERSION]) {

111

114

120

122  absolute_rho = false;
123

126 
127  if (options[WINT0])

131  // Look up desired model selection criterion using ar::best_model_function
132  // best_model_function template parameters fit ar::burg_method usage below
133  typedef ar::best_model_function<
134  ar::Burg,size_t,size_t,vector<real>
135  > best_model_function;
136  const best_model_function::type best_model
137  = best_model_function::lookup(criterion, subtract_mean);
138  if (!best_model) {
139  cerr << "Unknown model selection criterion: " << criterion << "\n";
140  return EXIT_FAILURE;
141  }
142 
143  // Use burg_method to estimate a hierarchy of AR models from input data
144  real mu;
145  vector<real> params, sigma2e, gain, autocor;
146  params .reserve(maxorder*(maxorder + 1)/2);
147  sigma2e.reserve(maxorder + 1);
148  gain .reserve(maxorder + 1);
149  autocor.reserve(maxorder + 1);
150  const size_t N = ar::burg_method(istream_iterator<real>(cin),
151  istream_iterator<real>(),
152  mu,
153  maxorder,
154  back_inserter(params),
155  back_inserter(sigma2e),
156  back_inserter(gain),
157  back_inserter(autocor),
158  subtract_mean,
159  /* output hierarchy? */ true);
160 
161 
162  // Keep only best model according to selected criterion
163  best_model(N, minorder, params, sigma2e, gain, autocor);
164 
165  // Compute decorrelation time from the estimated autocorrelation model
166  const real T0 = ar::decorrelation_time(
167  static_cast<size_t>(window_T0*N),
168  ar::autocorrelation(params.begin(), params.end(),
169  gain[0], autocor.begin()),
170  absolute_rho);
171 
172  // Compute effective variance and effective number of independent samples
173  const real eff_var = (N*gain[0]*sigma2e[0]) / (N - T0); // Trenberth1984
174  const real eff_N = N / T0;
175  const real mu_sigma = sqrt(eff_var / eff_N);
176 
177  // Output details about the best model and derived information
178  // Naming conventions here match the output of arsel-octfile by design
179  cout.precision(numeric_limits<real>::digits10 + 2);
180  cout << boolalpha
181  << "# absrho " << absolute_rho
182  << "\n# criterion " << criterion
183  << "\n# eff_N " << eff_N
184  << "\n# eff_var " << eff_var
185  << "\n# gain " << gain[0]
186  << "\n# maxorder " << maxorder
187  << "\n# minorder " << minorder
188  << "\n# mu " << mu
189  << "\n# mu_sigma " << mu_sigma
190  << "\n# N " << N
191  << "\n# AR(p) " << params.size()
192  << "\n# sigma2eps " << sigma2e[0]
193  << "\n# sigma2x " << gain[0]*sigma2e[0]
194  << "\n# submean " << subtract_mean
195  << "\n# T0 " << T0
196  << "\n# window_T0 " << window_T0
197  << noboolalpha
198  << showpos // Line up signs
199  << '\n' << real(1) // Leading one coefficient
200  << '\n';
201  copy(params.begin(), params.end(), ostream_iterator<real>(cout,"\n"));
202  cout.flush();
203 
204  return EXIT_SUCCESS;
205 }
