50"\n"
51" Automatically fit autoregressive models to input signals.\n"
52"\n"
53" Use ar::burg_method and ar::best_model to fit an autoregressive process\n"
54" for signals contained in the rows of matrix data. Sample means will\n"
55" be subtracted whenever submean is true. Model orders minorder through\n"
56" min(columns(data), maxorder) will be considered. A dictionary is\n"
57" returned where each key either contains a result indexable by the\n"

59" a single scalar applicable to all signals.\n"
60"\n"

62" Criteria are specified using the following abbreviations:\n"

64" AICC - asymptotically-corrected Akaike information criterion\n"

68" FSIC - finite sample information criterion\n"

70" MCC - minimally consistent criterion\n"

120extern "C" PyObject *ar_arsel(PyObject *self, PyObject *args, PyObject *kwargs)
121{
122 PyObject *ret_args = NULL, *ret = NULL;
123
124 // Sanity check that initar could build ar_ArselType
125 if (!PyType_Check(ar_ArselType)) {
126 PyErr_SetString(PyExc_RuntimeError,
127 "PyType_Check(ar_ArselType) failed");
128 return NULL;
129 }
130
131 // Prepare argument storage with default values
132 PyObject *data_obj = NULL;
133 int submean = DEFAULT_SUBMEAN;
134 int absrho = DEFAULT_ABSRHO;
135 const char *criterion = DEFAULT_CRITERION;
136 std::size_t minorder = DEFAULT_MINORDER;
137 std::size_t maxorder = DEFAULT_MAXORDER;
138
139 // Define keyword argument names
140 static char *kwlist[] = {
141 (char *)"data",
142 (char *)"submean",
143 (char *)"absrho",
144 (char *)"criterion",
145 (char *)"minorder",
146 (char *)"maxorder",
147 NULL
148 };
149
150 // Parse input tuple and keywords with second and subsequent arguments optional
151 {
152 unsigned long ul_minorder = minorder, ul_maxorder = maxorder;
153 if (!PyArg_ParseTupleAndKeywords(args, kwargs, "O|iiskk", kwlist,
154 &data_obj, &submean, &absrho,
155 &criterion,
156 &ul_minorder, &ul_maxorder)) {
157 return NULL;
158 }
159 minorder = ul_minorder;
160 maxorder = ul_maxorder;
161 }
162
163 // Lookup the desired model selection criterion
164 typedef ar::best_model_function<
165 ar::Burg,npy_intp,npy_intp,std::vector<double>
166 > best_model_function;
167 const best_model_function::type best_model
168 = best_model_function::lookup(std::string(criterion), submean);
169 if (!best_model) {
170 PyErr_SetString(PyExc_RuntimeError,
171 "Unknown model selection criterion provided to arsel.");
172 return NULL;
173 }
174
175 // Incoming data may be noncontiguous but should otherwise be well-behaved
176 // On success, 'data' is returned so Py_DECREF is applied only on failure
177 PyObject *data = PyArray_FROMANY(data_obj, NPY_DOUBLE, 1, 2,
178 NPY_ALIGNED | NPY_ELEMENTSTRIDES | NPY_NOTSWAPPED);
179 if (!data) {
180 Py_DECREF(data);
181 return NULL;
182 }
183
184 // Reshape any 1D ndarray into a 2D ndarray organized as a row vector.
185 // Permits the remainder of the routine to uniformly worry about 2D only.
186 if (1 == PyArray_NDIM((PyArrayObject *)data)) {
187 npy_intp dim[2] = { 1, PyArray_DIM((PyArrayObject *)data, 0) };
188 PyArray_Dims newshape = { dim, sizeof(dim)/sizeof(dim[0]) };
189 PyObject *olddata = data;
190 data = PyArray_Newshape((PyArrayObject *)olddata,
191 &newshape, NPY_ANYORDER);
192 Py_DECREF(olddata);
193 if (!data) {
194 Py_DECREF(data);
195 PyErr_SetString(PyExc_RuntimeError,
196 "Unable to reorganize data into matrix-like row vector.");
197 return NULL;
198 }
199 }
200
201 // How many data points are there?
202 npy_intp M = PyArray_DIM((PyArrayObject *)data, 0);
203 npy_intp N = PyArray_DIM((PyArrayObject *)data, 1);
204
205 // Prepare per-signal storage locations to return to caller
206 // TODO Ensure these invocations all worked as expected
207 PyObject *_AR = PyList_New(M);
208 PyObject *_autocor = PyList_New(M);
209 PyObject *_eff_N = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
210 PyObject *_eff_var = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
211 PyObject *_gain = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
212 PyObject *_mu = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
213 PyObject *_mu_sigma = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
214 PyObject *_sigma2eps = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
215 PyObject *_sigma2x = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
216 PyObject *_T0 = PyArray_ZEROS(1, &M, NPY_DOUBLE, 0);
217
218 // Prepare vectors to capture burg_method() output
219 std::vector<double> params, sigma2e, gain, autocor;
220 params .reserve(maxorder*(maxorder + 1)/2);
221 sigma2e.reserve(maxorder + 1);
222 gain .reserve(maxorder + 1);
223 autocor.reserve(maxorder + 1);
224
225 // Prepare repeatedly-used working storage for burg_method()
226 std::vector<double> f, b, Ak, ac;
227
228 // Process each signal in turn...
229 for (npy_intp i = 0; i < M; ++i)
230 {
231 // Use burg_method to estimate a hierarchy of AR models from input data
232 params .clear();
233 sigma2e.clear();
234 gain .clear();
235 autocor.clear();
236 ar::strided_adaptor<const double*> signal_begin(
237 (const double*) PyArray_GETPTR2((PyArrayObject *)data, i, 0),
238 PyArray_STRIDES((PyArrayObject *)data)[1] / sizeof(double));
239 ar::strided_adaptor<const double*> signal_end (
240 (const double*) PyArray_GETPTR2((PyArrayObject *)data, i, N),
241 PyArray_STRIDES((PyArrayObject *)data)[1] / sizeof(double));
242 ar::burg_method(signal_begin, signal_end,
243 *(double*)PyArray_GETPTR1((PyArrayObject *)_mu, i),
244 maxorder,
245 std::back_inserter(params),
246 std::back_inserter(sigma2e),
247 std::back_inserter(gain),
248 std::back_inserter(autocor),
249 submean, /* output hierarchy? */ true, f, b, Ak, ac);
250
251 // Keep only best model per chosen criterion via function pointer
252 try {
253 best_model(N, minorder, params, sigma2e, gain, autocor);
254 }
255 catch (std::exception &e)
256 {
257 PyErr_SetString(PyExc_RuntimeError, e.what());
258 goto fail;
259 }
260
261 // Compute decorrelation time from the estimated autocorrelation model
262 ar::predictor<double> p = ar::autocorrelation(
263 params.begin(), params.end(), gain[0], autocor.begin());
264 *(double*)PyArray_GETPTR1((PyArrayObject *)_T0, i)
265 = ar::decorrelation_time(N, p, absrho);
266
267 // Filter()-ready process parameters in field 'AR' with leading one
268 PyObject *_ARi = PyList_New(params.size() + 1);
269 PyList_SET_ITEM(_AR, i, _ARi);
270 PyList_SET_ITEM(_ARi, 0, PyFloat_FromDouble(1));
271 for (std::size_t k = 0; k < params.size(); ++k) {
272 PyList_SET_ITEM(_ARi, k+1, PyFloat_FromDouble(params[k]));
273 }
274
275 // Field 'sigma2eps'
276 *(double*)PyArray_GETPTR1((PyArrayObject *)_sigma2eps, i) = sigma2e[0];
277
278 // Field 'gain'
279 *(double*)PyArray_GETPTR1((PyArrayObject *)_gain, i) = gain[0];
280
281 // Field 'sigma2x'
282 *(double*)PyArray_GETPTR1((PyArrayObject *)_sigma2x, i) = gain[0]*sigma2e[0];
283
284 // Field 'autocor'
285 PyObject *_autocori = PyList_New(autocor.size());
286 PyList_SET_ITEM(_autocor, i, _autocori);
287 for (std::size_t k = 0; k < autocor.size(); ++k) {
288 PyList_SET_ITEM(_autocori, k, PyFloat_FromDouble(autocor[k]));
289 }
290
291 // Field 'eff_var'
292 // Unbiased effective variance expression from [Trenberth1984]
293 *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_var, i)
294 = (N*gain[0]*sigma2e[0]) / (N - *(double*)PyArray_GETPTR1((PyArrayObject *)_T0, i));
295
296 // Field 'eff_N'
297 *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_N, i)
298 = N / *(double*)PyArray_GETPTR1((PyArrayObject *)_T0, i);
299
300 // Field 'mu_sigma'
301 // Variance of the sample mean using effective quantities
302 *(double*)PyArray_GETPTR1((PyArrayObject *)_mu_sigma, i)
303 = std::sqrt( *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_var, i)
304 / *(double*)PyArray_GETPTR1((PyArrayObject *)_eff_N, i));
305 }
306
307 // Prepare build and return an ar_ArselType via tuple constructor
308 // See initar(...) method for the collections.namedtuple-based definition
309 ret_args = PyTuple_Pack(17, PyBool_FromLong(absrho),
310 _AR,
311 _autocor,
312 PyUnicode_FromString(criterion),
313 data,
314 _eff_N,
315 _eff_var,
316 _gain,
317 PyLong_FromSize_t(maxorder),
318 PyLong_FromSize_t(minorder),
319 _mu,
320 _mu_sigma,
321 PyLong_FromLong(N),
322 _sigma2eps,
323 _sigma2x,
324 PyBool_FromLong(submean),
325 _T0);
326 if (!ret_args) {
327 PyErr_SetString(PyExc_RuntimeError,
328 "Unable to prepare arguments used to build return value.");
329 goto fail;
330 }
331 ret = PyObject_CallObject((PyObject *)ar_ArselType, ret_args);
332 if (!ret) {
333 PyErr_SetString(PyExc_RuntimeError,
334 "Unable to construct return value.");
335 goto fail;
336 }
337
338 return ret;
339
340fail:
341 Py_XDECREF(ret_args);
342 Py_XDECREF(ret);
343 Py_XDECREF(data);
344 Py_XDECREF(_AR);
345 Py_XDECREF(_autocor);
346 Py_XDECREF(_eff_N);
347 Py_XDECREF(_eff_var);
348 Py_XDECREF(_gain);
349 Py_XDECREF(_mu);
350 Py_XDECREF(_mu_sigma);
351 Py_XDECREF(_sigma2eps);
352 Py_XDECREF(_sigma2x);
353 Py_XDECREF(_T0);
354 return NULL;
355}

434extern "C" PyMODINIT_FUNC PyInit_ar(void)
435{
436 PyObject *module = PyModule_Create(&moduledef);
437
438 if (module == NULL)
439 INITERROR;
440 struct module_state *st = GETSTATE(module);
441
442 st->error = PyErr_NewException("ar.Error", NULL, NULL);
443 if (st->error == NULL) {
444 Py_DECREF(module);
445 INITERROR;
446 }
447
448 // Register NumPy and the Arsel namedtuple
449 import_array();
450 init_namedtuple();
