175template <typename InputIterator,
176 typename OutputType1,
177 typename OutputType2>
178std::size_t welford_nvariance(InputIterator first,
179 InputIterator last,
180 OutputType1& mean,
181 OutputType2& nvar)
182{
183 using std::iterator_traits;
184 using std::size_t;
185 typedef typename iterator_traits<InputIterator>::value_type value;
186
187 size_t N = 1; // Running next sample number
188 value m = 0; // Running mean of data thus far
189 value nv = 0; // Running variance times the number of samples
190
191 while (first != last)
192 {
193 value x = *first++;
194 value d = x - m;
195 m += d / N++;
196 nv += d*(x - m);
197 }
198
199 mean = m;
200 nvar = nv;
201 return N-1;
202}

266template <typename InputIterator1,
267 typename InputIterator2,
268 typename OutputType1,
269 typename OutputType2,
270 typename OutputType3>
271std::size_t welford_ncovariance(InputIterator1 first1,
272 InputIterator1 last1,
273 InputIterator2 first2,
274 OutputType1& mean1,
275 OutputType2& mean2,
276 OutputType3& ncovar)
277{
278 using std::iterator_traits;
279 using std::size_t;
280 typedef typename iterator_traits<InputIterator1>::value_type value1;
281 typedef typename iterator_traits<InputIterator2>::value_type value2;
282
283 size_t N = 1; // Running next sample number
284 value1 m1 = 0; // Running mean of first data set thus far
285 value2 m2 = 0; // Running mean of second data set thus far
286 OutputType3 nc = 0; // Running covariance times the number of samples
287
288 while (first1 != last1)
289 {
290 value1 x1 = *first1++;
291 value1 d1 = x1 - m1;
292 m1 += d1 / N;
293
294 value2 x2 = *first2++;
295 value2 d2 = x2 - m2;
296 m2 += d2 / N;
297
298 nc += d1*(x2 - m2);
299
300 ++N;
301 }
302
303 mean1 = m1;
304 mean2 = m2;
305 ncovar = nc;
306 return N-1;
307}

463template <typename ValueType,
464 typename InputIterator1,
465 typename InputIterator2>
466ValueType
467#if (AR_GCC_VERSION > 40305)
468 __attribute__((__optimize__("no-associative-math")))
469#endif
470negative_half_reflection_coefficient(InputIterator1 a_first,
471 InputIterator1 a_last,
472 InputIterator2 b_first)
473#if (AR_GCC_VERSION > 40305) || (_MSC_VER > 1400)
474{
475 ValueType ns = 0, nt, nc = 0, ny; // Kahan numerator accumulation
476 ValueType ds = 0, dt, dc = 0, dy; // Kahan denominator accumulation
477
478 while (a_first != a_last)
479 {
480 ValueType xa = *a_first++; // Denominator: \vec{a}\cdot\vec{a}
481 dy = (xa * xa) - dc;
482 dt = ds + dy;
483 dc = (dt - ds) - dy;
484 ds = dt;
485
486 ValueType xb = *b_first++; // Denominator: \vec{b}\cdot\vec{b}
487 dy = (xb * xb) - dc;
488 dt = ds + dy;
489 dc = (dt - ds) - dy;
490 ds = dt;
491
492 ny = (xa * xb) - nc; // Numerator: \vec{a}\cdot\vec{b}
493 nt = ns + ny;
494 nc = (nt - ns) - ny;
495 ns = nt;
496 }
497
498 return ns + nc == 0 // Does special zero case apply?
499 ? 0 // Yes, to avoid NaN from 0 / 0
500 : (ns + nc) / (ds + dc); // No, correct final sums and form ratio
501}
502#else
503#warning Using Non-Kahan version of ar::negative_half_reflection_coefficient.
504{
505 ValueType ns = 0;
506 ValueType ds = 0;
507
508 while (a_first != a_last)
509 {
510 ValueType xa = *a_first++;
511 ValueType xb = *b_first++;
512 ns += xa * xb; // Numerator
513 ds += xa * xa + xb * xb; // Denominator
514 }
515
516 return ns == 0 // Does special zero case apply?
517 ? 0 // Yes, to avoid NaN from 0 / 0
518 : ns / ds; // No, form ratio
519}
520#endif

616template <class InputIterator,
617 class Value,
618 class OutputIterator1,
619 class OutputIterator2,
620 class OutputIterator3,
621 class OutputIterator4,
622 class Vector>
623std::size_t burg_method(InputIterator data_first,
624 InputIterator data_last,
625 Value& mean,
626 std::size_t& maxorder,
627 OutputIterator1 params_first,
628 OutputIterator2 sigma2e_first,
629 OutputIterator3 gain_first,
630 OutputIterator4 autocor_first,
631 const bool subtract_mean,
632 const bool hierarchy,
633 Vector& f,
634 Vector& b,
635 Vector& Ak,
636 Vector& ac)
637{
638 using std::bind2nd;
639 using std::copy;
640 using std::distance;
641 using std::fill;
642 using std::inner_product;
643 using std::min;
644 using std::minus;
645 using std::size_t;
646
647 // Initialize f from [data_first, data_last) and fix number of samples
648 f.assign(data_first, data_last);
649 const size_t N = f.size();
650
651 // Stably compute the incoming data's mean and population variance
652 mean = 0;
653 Value sigma2e = 0;
654 welford_variance_population(f.begin(), f.end(), mean, sigma2e);
655
656 // When requested, subtract the just-computed mean from the data.
657 // Adjust, if necessary, to make sigma2e the second moment.
658 if (subtract_mean)
659 {
660 for (typename Vector::iterator it = f.begin(); it != f.end(); ++it) {
661 *it -= mean;
662 }
663 }
664 else
665 {
666 sigma2e += mean*mean;
667 }
668
669 // At most maxorder N-1 can be fit from N samples. Beware N is unsigned.
670 maxorder = (N == 0) ? 0 : min(static_cast<size_t>(maxorder), N-1);
671
672 // Output sigma2e and gain for a zeroth order model, if requested.
673 Value gain = 1;
674 if (hierarchy || maxorder == 0)
675 {
676 *sigma2e_first++ = sigma2e;
677 *gain_first++ = gain;
678 }
679
680 // Initialize and perform Burg recursion
681 if (maxorder) b = f; // Copy iff non-trivial work required
682 Ak.assign(maxorder + 1, Value(0));
683 Ak[0] = 1;
684 ac.clear();
685 ac.reserve(maxorder);
686 for (size_t kp1 = 1; kp1 <= maxorder; ++kp1)
687 {
688 // Compute mu from f, b, and Dk and then update sigma2e and Ak using mu
689 // Afterwards, Ak[1:kp1] contains AR(k) coefficients by the recurrence
690 // Must treat mu result of 0 / 0 as 0 to avoid NaNs on constant signals
691 // By the recurrence, Ak[kp1] will also be the reflection coefficient
692 Value mu = -2 * negative_half_reflection_coefficient<Value>(
693 f.begin() + kp1, f.end(), b.begin());
694
695 sigma2e *= (1 - mu*mu);
696 for (size_t n = 0; n <= kp1/2; ++n)
697 {
698 Value t1 = Ak[n] + mu*Ak[kp1 - n];
699 Value t2 = Ak[kp1 - n] + mu*Ak[n];
700 Ak[n] = t1;
701 Ak[kp1 - n] = t2;
702 }
703
704 // Update the gain per Broersen 2006 equation (5.25)
705 gain *= 1 / (1 - Ak[kp1]*Ak[kp1]);
706
707 // Compute and output the next autocorrelation coefficient
708 // See Broersen 2006 equations (5.28) and (5.31) for details
709 ac.push_back(-inner_product(ac.rbegin(), ac.rend(),
710 Ak.begin() + 1, Ak[kp1]));
711
712 // Output parameters and the input and output variances when requested
713 if (hierarchy || kp1 == maxorder)
714 {
715 params_first = copy(Ak.begin() + 1, Ak.begin() + kp1 + 1,
716 params_first);
717 *sigma2e_first++ = sigma2e;
718 *gain_first++ = gain;
719 }
720
721 // Update f and b for the next iteration if another remains
722 if (kp1 < maxorder)
723 {
724 for (size_t n = 0; n < N - kp1; ++n)
725 {
726 Value t1 = f[n + kp1] + mu*b[n];
727 Value t2 = b[n] + mu*f[n + kp1];
728 f[n + kp1] = t1;
729 b[n] = t2;
730 }
731 }
732 }
733
734 // Output the lag [0,maxorder] autocorrelation coefficients in single pass
735 *autocor_first++ = 1;
736 copy(ac.begin(), ac.end(), autocor_first);
737
738 // Return the number of values processed in [data_first, data_last)
739 return N;
740}

947 template <class InputIterator>
948 predictor& initial_conditions(InputIterator initial_first,
949 const Value x0adjust = 0)
950 {
951 // Zero the simulation time.
952 n = 0;
953
954 // Set d = [ a_p, ..., a_1, x_{n-p}, ..., x_{n-1} ]
955 using std::vector;
956 typename vector<Value>::size_type i = d.size();
957 typename vector<Value>::size_type p = i / 2;
958 while (i --> p) d[i] = *initial_first++;
959
960 // Make x_n := - a_p*x_{n-p} - ... - a_1*x_{n-1} + x_n + x0adjust.
961 // By design, x_n was whatever it happened to be.
962 using std::inner_product;
963 xn += x0adjust;
964 xn = -inner_product(d.begin(), d.begin() + p, d.begin() + p, -xn);
965
966 return *this;
967 }
968
969 // Concept: InputIterator
970
972 predictor& operator++()
973 {
974 using std::distance;
975 using std::inner_product;
976 using std::vector;
977
978 if (g)
979 {
980 typename vector<Value>::size_type p = d.size() / 2;
981 if (p)
982 {
983 // Make x_n = - a_p*x_{n-p} - ... - a_1*x_{n-1} + \epsilon_n
984 // by (conceptually) storing previously computed x_n into
985 // circular buffer, updating ++n, and computing x_{n+1}.
986 typename vector<Value>::iterator ab = d.begin();
987 typename vector<Value>::iterator xb = ab + p;
988 typename vector<Value>::iterator c = xb + n % p;
989 typename vector<Value>::iterator xe = d.end();
990 *c++ = xn;
991 xn = inner_product(c, xe, ab, -(*g)());
992 xn = -inner_product(xb, c, ab + distance(c, xe), xn );
993 }
994 else
995 {
996 xn = (*g)();
997 }
998 }
999 else
1000 {
1001#ifndef NDEBUG
1002 using std::numeric_limits;
1003 if (numeric_limits<Value>::has_quiet_NaN)
1004 xn = numeric_limits<Value>::quiet_NaN();
1005#endif
1006 }
1007
1008 ++n;
1009
1010 return *this;
1011 }

1235template<class RandomAccessIterator,
1236 class InputIterator,
1237 class OutputIterator>
1238void zohar_linear_solve(RandomAccessIterator a_first,
1239 RandomAccessIterator a_last,
1240 RandomAccessIterator r_first,
1241 InputIterator d_first,
1242 OutputIterator s_first)
1243{
1244 using std::copy;
1245 using std::distance;
1246 using std::inner_product;
1247 using std::invalid_argument;
1248 using std::iterator_traits;
1249 using std::reverse_iterator;
1250 using std::vector;
1251
1252 // Tildes indicate transposes while hats indicate reversed vectors.
1253
1254 // InputIterator::value_type determines the working precision
1255 typedef typename iterator_traits<InputIterator>::value_type value_type;
1256 typedef vector<value_type> vector_type;
1257 typedef typename vector_type::size_type size_type;
1258
1259 // Determine problem size using [a_first,a_last) and ensure nontrivial
1260 typename iterator_traits<RandomAccessIterator>::difference_type dist
1261 = distance(a_first, a_last);
1262 if (dist < 1) throw invalid_argument("distance(a_first, a_last) < 1");
1263 const size_type n = static_cast<size_type>(dist);
1264
1265 // Allocate working storage and set initial values for recursion:
1266 vector_type s; s .reserve(n+1); s .push_back( *d_first);
1267 vector_type ehat; ehat.reserve(n ); ehat.push_back(-a_first[0]);
1268 vector_type g; g .reserve(n ); g .push_back(-r_first[0]);
1269 value_type lambda = 1 - a_first[0]*r_first[0];
1270
1271 // Though recursive updates to s and g can be done in-place, updates to
1272 // ehat seemingly require one additional vector for storage:
1273 //
1274 // "This sequence of computations is economical of storage. It is only
1275 // necessary to retain quantities computed at level m - 1 until the
1276 // computations at level m are complete." [Trench1967, page 1504]
1277 vector_type next_ehat; next_ehat.reserve(n);
1278
1279 // Recursion for i = {1, 2, ..., n - 1}:
1280 for (size_type i = 1; i < n; ++i)
1281 {
1282
1283 reverse_iterator<RandomAccessIterator> rhat_first(r_first + i);
1284
1285 // \theta_i = \delta_{i+1} - \tilde{s}_i \hat{r}_i
1286 const value_type neg_theta = inner_product(
1287 s.begin(), s.end(), rhat_first, value_type(-(*++d_first)));
1288
1289 // \eta_i = -\rho_{-(i+1)} - \tilde{a}_i \hat{e}_i
1290 const value_type neg_eta = inner_product(
1291 ehat.begin(), ehat.end(), a_first, value_type(a_first[i]));
1292
1293 // \gamma_i = -\rho_{i+1} - \tilde{g}_i \hat{r}_i
1294 const value_type neg_gamma = inner_product(
1295 g.begin(), g.end(), rhat_first, value_type(r_first[i]));
1296
1297 /*
1298 * s_{i+1} = \bigl(\begin{smallmatrix}
1299 * s_i + (\theta_i/\lambda_i) \hat{e}_i \\
1300 * \theta_i/\lambda_i
1301 * \end{smallmatrix}\bigr)
1302 *
1303 * \hat{e}_{i+1} = \bigl(\begin{smallmatrix}
1304 * \eta_i/\lambda_i \\
1305 * \hat{e}_i + (\eta_i/\lambda_i) g_i
1306 * \end{smallmatrix}\bigr)
1307 *
1308 * g_{i+1} = \bigl(\begin{smallmatrix}
1309 * g_i + (\gamma_i/\lambda_i) \hat{e}_i \\
1310 * \gamma_i/\lambda_i
1311 * \end{smallmatrix}\bigr)
1312 */
1313 const value_type theta_by_lambda = -neg_theta/lambda;
1314 const value_type eta_by_lambda = -neg_eta /lambda;
1315 const value_type gamma_by_lambda = -neg_gamma/lambda;
1316 next_ehat.clear();
1317 next_ehat.push_back(eta_by_lambda);
1318 for (size_type j = 0; j < i; ++j)
1319 {
1320 s[j] += theta_by_lambda*ehat[j];
1321 next_ehat.push_back(ehat[j] + eta_by_lambda*g[j]);
1322 g[j] += gamma_by_lambda*ehat[j];
1323 }
1324 s.push_back(theta_by_lambda);
1325 g.push_back(gamma_by_lambda);
1326 ehat.swap(next_ehat);
1327
1328 // \lambda_{i+1} = \lambda_i - \eta_i \gamma_i / \lambda_i
1329 lambda -= neg_eta*neg_gamma/lambda;
1330 }
1331
1332 // Recursion for i = n differs slightly per Zohar's "Last Computed Values"
1333 // Computing g_n above was unnecessary but the incremental expense is small
1334 {
1335 reverse_iterator<RandomAccessIterator> rhat_first(r_first + n);
1336
1337 // \theta_n = \delta_{n+1} - \tilde{s}_n \hat{r}_n
1338 const value_type neg_theta = inner_product(
1339 s.begin(), s.end(), rhat_first, value_type(-(*++d_first)));
1340
1341 /*
1342 * s_{n+1} = \bigl(\begin{smallmatrix}
1343 * s_n + (\theta_n/\lambda_n) \hat{e}_n \\
1344 * \theta_n/\lambda_n
1345 * \end{smallmatrix}\bigr)
1346 */
1347 const value_type theta_by_lambda = -neg_theta/lambda;
1348 for (size_type j = 0; j < n; ++j)
1349 {
1350 s[j] += theta_by_lambda*ehat[j];
1351 }
1352 s.push_back(theta_by_lambda);
1353 }
1354
1355 // Output solution
1356 copy(s.begin(), s.end(), s_first);
1357}
