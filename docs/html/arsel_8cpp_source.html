71int main(int argc, char *argv[])
72{
73 using namespace std;
74
75 // Parse and process any command line arguments using optionparser.h
76 string criterion = "CIC";
77 bool subtract_mean = false;
78 size_t minorder = 0;
79 size_t maxorder = 512;
80 bool absolute_rho = true;
81 double window_T0 = 1.0;
82 {
83 option::Stats stats(usage, argc-(argc>0), argv+(argc>0));
84
85 vector<option::Option> options(stats.options_max + stats.buffer_max);
86
87 option::Parser parse(usage, argc-(argc>0), argv+(argc>0),
88 &options[0], &options[stats.options_max]);
89
90 if (parse.error() || options[UNKNOWN]) {
91 for (option::Option* o = options[UNKNOWN]; o; o = o->next()) {
92 cerr << "Unknown option: " << o->name << "\n";
93 }
94 return EXIT_FAILURE;
95 }
96
97 if (options[HELP]) {
98 printUsage(cout, usage);
99 return EXIT_SUCCESS;
100 }
101
102 if (options[VERSION]) {
103 cout << "arsel " << ARSEL_VERSION << "\n";
104 return EXIT_SUCCESS;
105 }
106
107 if (options[CRITERION])
108 criterion = options[CRITERION].last()->arg;
109
110 if (options[MAXORDER])
111 maxorder = (size_t) strtol(options[MAXORDER].last()->arg, NULL, 10);
112
113 if (options[MINORDER])
114 minorder = (size_t) strtol(options[MINORDER].last()->arg, NULL, 10);
115
116 if (options[NONABSRHO])
117 absolute_rho = false;
118
119 if (options[SUBMEAN])
120 subtract_mean = true;
121
122 if (options[WINT0])
123 window_T0 = strtod(options[WINT0].last()->arg, NULL);
124 }
125
126 // Look up desired model selection criterion using ar::best_model_function
127 // best_model_function template parameters fit ar::burg_method usage below
128 typedef ar::best_model_function<
129 ar::Burg,size_t,size_t,vector<real>
130 > best_model_function;
131 const best_model_function::type best_model
132 = best_model_function::lookup(criterion, subtract_mean);
133 if (!best_model) {
134 cerr << "Unknown model selection criterion: " << criterion << "\n";
135 return EXIT_FAILURE;
136 }
137
138 // Use burg_method to estimate a hierarchy of AR models from input data
139 real mu;
140 vector<real> params, sigma2e, gain, autocor;
141 params .reserve(maxorder*(maxorder + 1)/2);
142 sigma2e.reserve(maxorder + 1);
143 gain .reserve(maxorder + 1);
144 autocor.reserve(maxorder + 1);
145 const size_t N = ar::burg_method(istream_iterator<real>(cin),
146 istream_iterator<real>(),
147 mu,
148 maxorder,
149 back_inserter(params),
150 back_inserter(sigma2e),
151 back_inserter(gain),
152 back_inserter(autocor),
153 subtract_mean,
154 /* output hierarchy? */ true);
155
156
157 // Keep only best model according to selected criterion
158 best_model(N, minorder, params, sigma2e, gain, autocor);
159
160 // Compute decorrelation time from the estimated autocorrelation model
161 const real T0 = ar::decorrelation_time(
162 static_cast<size_t>(window_T0*N),
163 ar::autocorrelation(params.begin(), params.end(),
164 gain[0], autocor.begin()),
165 absolute_rho);
166
167 // Compute effective variance and effective number of independent samples
168 const real eff_var = (N*gain[0]*sigma2e[0]) / (N - T0); // Trenberth1984
169 const real eff_N = N / T0;
170 const real mu_sigma = sqrt(eff_var / eff_N);
171
172 // Output details about the best model and derived information
173 // Naming conventions here match the output of arsel-octfile by design
174 cout.precision(numeric_limits<real>::digits10 + 2);
175 cout << boolalpha
176 << "# absrho " << absolute_rho
177 << "\n# criterion " << criterion
178 << "\n# eff_N " << eff_N
179 << "\n# eff_var " << eff_var
180 << "\n# gain " << gain[0]
181 << "\n# maxorder " << maxorder
182 << "\n# minorder " << minorder
183 << "\n# mu " << mu
184 << "\n# mu_sigma " << mu_sigma
185 << "\n# N " << N
186 << "\n# AR(p) " << params.size()
187 << "\n# sigma2eps " << sigma2e[0]
188 << "\n# sigma2x " << gain[0]*sigma2e[0]
189 << "\n# submean " << subtract_mean
190 << "\n# T0 " << T0
191 << "\n# window_T0 " << window_T0
192 << noboolalpha
193 << showpos // Line up signs
194 << '\n' << real(1) // Leading one coefficient
195 << '\n';
196 copy(params.begin(), params.end(), ostream_iterator<real>(cout,"\n"));
197 cout.flush();
198
199 return EXIT_SUCCESS;
200}
