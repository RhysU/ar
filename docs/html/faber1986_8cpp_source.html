43static void faber1986(real data[],
44 int ndat,
45 real k[],
46 real a[],
47 real err[],
48 int p)
49{
50 int i; /* order index, i <= i <= p */
51 int j; /* order sub-index, j < i */
52 int n; /* time index, 0 <= n < ndat */
53 real e[MAXSIZE]; /* forward prediction error[n] */
54 real b[MAXSIZE]; /* backward pred. error[n] */
55 real den; /* denominator, eq. (2.74) */
56 real num; /* numerator, eq. (2.73) */
57 real dk; /* holder for k[i] */
58 real ta[MAXORD]; /* temporary holder for a[i] */
59
60 /******* Initialize *******/
61 a[0] = 1;
62 err[0] = k[0] = dk = 0;
63 for (n = 0; n < ndat; n++) {
64 e[n] = b[n] = data[n];
65 err[0] += data[n] * data[n];
66 }
67 den = 2 * err[0];
68
69 /******* Order Recursion *******/
70 for (i = 1; i <= p; i++) {
71 /**** Compute denom., corrected eq. (2.74) ****/
72 den = den * (1 - dk * dk) - e[i-1] * e[i-1]
73 - b[ndat - 1] * b[ndat - 1];
74 /**** Compute k[i] eq. (2.73) ****/
75 num = 0;
76 for (n = i; n < ndat; n++)
77 num += b[n-1] * e[n];
78 num *= -2;
79 k[i] = dk = num / den;
80 /**** Update b[n], e[n], eq. (2.54), (2.52) ****/
81 for (n=ndat - 1; n >= i; n--) {
82 b[n] = b[n-1] + dk * e[n];
83 e[n] = e[n] + dk * b[n-1];
84 }
85 /**** Update a[j], eq. (2.45) ****/
86 for (j=1; j<i; j++)
87 ta[j] = a[j] + dk*a[i-j];
88 for (j=1; j<i; j++)
89 a[j]=ta[j];
90 a[i] = dk;
91 /**** Update err[i], eq. (2.46) ****/
92 err[i] = err[i-1] * (1 - dk * dk);
93 }
94} /* ends faber1986 */
