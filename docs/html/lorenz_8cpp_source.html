102static inline void lorenz(
103 const real beta, const real rho, const real sigma,
104 const real x, const real y, const real z,
105 real &dxdt, real &dydt, real &dzdt)
106{
107 dxdt = sigma*(y - x);
108 dydt = x*(rho - z) - y;
109 dzdt = x*y - beta*z;
110}
111
113static void euler(
114 const real dt, const real beta, const real rho, const real sigma,
115 long double &t, real &x, real &y, real &z)
116{
117 real dxdt, dydt, dzdt;
118 lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
119 x += dt*dxdt;
120 y += dt*dydt;
121 z += dt*dzdt;
122 t += dt;
123}

129static void tvd_rk2(
130 const real dt, const real beta, const real rho, const real sigma,
131 long double &t, real &x, real &y, real &z)
132{
133 real dxdt, dydt, dzdt;
134
135 lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
136 real u1x = x + dt*dxdt;
137 real u1y = y + dt*dydt;
138 real u1z = z + dt*dzdt;
139
140 lorenz(beta, rho, sigma, u1x, u1y, u1z, dxdt, dydt, dzdt);
141 x = (x + u1x + dt*dxdt)/2;
142 y = (y + u1y + dt*dydt)/2;
143 z = (z + u1z + dt*dzdt)/2;
144 t += dt;
145}

151static void tvd_rk3(
152 const real dt, const real beta, const real rho, const real sigma,
153 long double &t, real &x, real &y, real &z)
154{
155 real dxdt, dydt, dzdt;
156
157 lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
158 real u1x = x + dt*dxdt;
159 real u1y = y + dt*dydt;
160 real u1z = z + dt*dzdt;
161
162 lorenz(beta, rho, sigma, u1x, u1y, u1z, dxdt, dydt, dzdt);
163 real u2x = (3*x + u1x + dt*dxdt)/4;
164 real u2y = (3*y + u1y + dt*dydt)/4;
165 real u2z = (3*z + u1z + dt*dzdt)/4;
166
167 lorenz(beta, rho, sigma, u2x, u2y, u2z, dxdt, dydt, dzdt);
168 x = (x + 2*u2x + 2*dt*dxdt)/3;
169 y = (y + 2*u2y + 2*dt*dydt)/3;
170 z = (z + 2*u2z + 2*dt*dzdt)/3;
171 t += dt;
172}

177static void std_rk4(
178 const real dt, const real beta, const real rho, const real sigma,
179 long double &t, real &x, real &y, real &z)
180{
181 real dxdt, dydt, dzdt;
182
183 lorenz(beta, rho, sigma, x, y, z, dxdt, dydt, dzdt);
184 real ux = x + dt*dxdt/2;
185 real uy = y + dt*dydt/2;
186 real uz = z + dt*dzdt/2;
187
188 real dx = dt*dxdt/6;
189 real dy = dt*dydt/6;
190 real dz = dt*dzdt/6;
191
192 lorenz(beta, rho, sigma, ux, uy, uz, dxdt, dydt, dzdt);
193 ux = x + dt*dxdt/2;
194 uy = y + dt*dydt/2;
195 uz = z + dt*dzdt/2;
196
197 dx += dt*dxdt/3;
198 dy += dt*dydt/3;
199 dz += dt*dzdt/3;
200
201 lorenz(beta, rho, sigma, ux, uy, uz, dxdt, dydt, dzdt);
202 ux = x + dt*dxdt;
203 uy = y + dt*dydt;
204 uz = z + dt*dzdt;
205
206 dx += dt*dxdt/3;
207 dy += dt*dydt/3;
208 dz += dt*dzdt/3;
209
210 lorenz(beta, rho, sigma, ux, uy, uz, dxdt, dydt, dzdt);
211 x += (dx + dt*dxdt/6);
212 y += (dy + dt*dydt/6);
213 z += (dz + dt*dzdt/6);
214 t += dt;
215}

228int main(int argc, char *argv[])
229{
230 using namespace std;
231
232 real beta = 8./3;
233 real burn = 500;
234 real dt = 0.01;
235 long every = 1;
236 real rho = 28;
237 real sigma = 10;
238 long double t = 0;
239 real tfinal = 3000;
240 bool more = false;
241 SchemeType scheme;
242 real x;
243 real y;
244 real z;
245 {
246 option::Stats stats(usage, argc-(argc>0), argv+(argc>0));
247
248 vector<option::Option> opts(stats.options_max + stats.buffer_max);
249
250 option::Parser parse(usage, argc-(argc>0), argv+(argc>0),
251 &opts[0], &opts[stats.options_max]);
252
253 if (parse.error() || opts[UNKNOWN]) {
254 for (option::Option* o = opts[UNKNOWN]; o; o = o->next()) {
255 cerr << "Unknown option: " << o->name << "\n";
256 }
257 return EXIT_FAILURE;
258 }
259
260 if (opts[HELP]) {
261 printUsage(cout, usage);
262 return EXIT_SUCCESS;
263 }
264
265 // SEED must come before any other randomly-generated option, e.g. INITX
266 if (opts[SEED]) {
267 srandom((unsigned) strtol(opts[SEED].last()->arg, NULL, 10));
268 } else {
269 struct timeval tv;
270 struct timezone tz;
271 gettimeofday(&tv, &tz);
272 srand((unsigned) tv.tv_usec);
273 }
274
275 // INITX, INITY, INITZ either specified or randomized
276 x = opts[INITX] ? strtod(opts[INITX].last()->arg, NULL)
277 : (real)rand()/(real)RAND_MAX;
278 y = opts[INITY] ? strtod(opts[INITY].last()->arg, NULL)
279 : (real)rand()/(real)RAND_MAX;
280 z = opts[INITZ] ? strtod(opts[INITZ].last()->arg, NULL)
281 : (real)rand()/(real)RAND_MAX;
282
283 // Default timestepping scheme is whichever SchemeType has value zero
284 scheme = static_cast<SchemeType>(opts[SCHEME].last()->type());
285
286 // Parse remaining options
287 if (opts[BETA ]) beta = strtod(opts[BETA ].last()->arg, NULL);
288 if (opts[BURN ]) burn = strtod(opts[BURN ].last()->arg, NULL);
289 if (opts[DT ]) dt = strtod(opts[DT ].last()->arg, NULL);
290 if (opts[EVERY ]) every = strtol(opts[EVERY ].last()->arg, NULL, 10);
291 if (opts[RHO ]) rho = strtod(opts[RHO ].last()->arg, NULL);
292 if (opts[SIGMA ]) sigma = strtod(opts[SIGMA ].last()->arg, NULL);
293 if (opts[TFINAL]) tfinal = strtod(opts[TFINAL].last()->arg, NULL);
294 if (opts[MORE ]) more = true;
295
296 // Warn whenever burn >= tfinal
297 if (burn >= tfinal) {
298 cerr << "Warning: burn >= tfinal suppresses output ("
299 << burn << " >= " << tfinal << ") \n";
300 }
301 }
302
303 // Discard 0 <= t < burn
304 switch (scheme) { // Switch designed to avoid spurious jumps
305 default: cerr << "Sanity failure: unknown scheme at "
306 << __FILE__ << ":" << __LINE__ << '\n';
307 return EXIT_FAILURE;
308 case RK1: while (t < burn) euler (dt, beta, rho, sigma, t, x, y, z);
309 break;
310 case RK2: while (t < burn) tvd_rk2(dt, beta, rho, sigma, t, x, y, z);
311 break;
312 case RK3: while (t < burn) tvd_rk3(dt, beta, rho, sigma, t, x, y, z);
313 break;
314 case RK4: while (t < burn) std_rk4(dt, beta, rho, sigma, t, x, y, z);
315 break;
316 }
317
318 // Output (t, x, y, z) during burn < t <= tfinal at periodic intervals
319 cout.precision(numeric_limits<real>::digits10 + 2);
320 cout << showpos;
321 do {
322
323 cout << t << '\t' << x << '\t' << y << '\t' << z;
324 if (more) {
325 cout << '\t' << x*x << '\t' << x*y << '\t' << x*z
326 << '\t' << y*y << '\t' << y*z
327 << '\t' << z*z;
328 }
329 cout << '\n';
330
331 switch (scheme) {
332 default:
333 cerr << "Sanity failure: unknown scheme at "
334 << __FILE__ << ":" << __LINE__ << '\n';
335 return EXIT_FAILURE;
336 case RK1:
337 for (long i = 0; i < every; ++i) {
338 euler (dt, beta, rho, sigma, t, x, y, z);
339 }
340 break;
341 case RK2:
342 for (long i = 0; i < every; ++i) {
343 tvd_rk2(dt, beta, rho, sigma, t, x, y, z);
344 }
345 break;
346 case RK3:
347 for (long i = 0; i < every; ++i) {
348 tvd_rk3(dt, beta, rho, sigma, t, x, y, z);
349 }
350 break;
351 case RK4:
352 for (long i = 0; i < every; ++i) {
353 std_rk4(dt, beta, rho, sigma, t, x, y, z);
354 }
355 break;
356 }
357
358 } while (t < tfinal);
359
360 return EXIT_SUCCESS;
361}
