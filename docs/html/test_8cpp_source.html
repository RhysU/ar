44// Computes percent difference of \c b against theoretical result \c a.

51// Test burg_method against synthetic data
52int main(int argc, char *argv[])
53{
54 using namespace ar;
55 using namespace std;
56
57 string filename_coeffs, filename_data;
58 bool subtract_mean = false;
59 {
60 option::Stats stats(usage, argc-(argc>0), argv+(argc>0));
61
62 vector<option::Option> options(stats.options_max + stats.buffer_max);
63
64 option::Parser parse(usage, argc-(argc>0), argv+(argc>0),
65 &options[0], &options[stats.options_max]);
66
67 if (parse.error() || options[UNKNOWN]) {
68 for (option::Option* o = options[UNKNOWN]; o; o = o->next()) {
69 cerr << "Unknown option: " << o->name << "\n";
70 }
71 return EXIT_FAILURE;
72 }
73
74 if (options[HELP]) {
75 printUsage(cout, usage);
76 return EXIT_SUCCESS;
77 }
78
79 if (options[SUBMEAN])
80 subtract_mean = true;
81
82 if (parse.nonOptionsCount() != 2) {
83 cerr << "Missing one or more file operands. Try --help.\n";
84 return EXIT_FAILURE;
85 }
86 filename_coeffs = parse.nonOption(0);
87 filename_data = parse.nonOption(1);
88 }
89
90 // Read "exact" coefficients
91 vector<real> exact;
92 {
93 ifstream f;
94 f.exceptions(ifstream::badbit);
95 f.open(filename_coeffs.c_str());
96 copy(istream_iterator<real>(f), istream_iterator<real>(),
97 back_inserter(exact));
98 }
99 vector<real> est(exact.size()), cor(exact.size() + 1);
100
101 // Read time series data
102 ifstream f;
103 f.exceptions(ifstream::badbit);
104 f.open(filename_data.c_str());
105
106 // Use burg_method to fit an AR model and characterize it completely
107 size_t maxorder = exact.size();
108 real mean, sigma2e, gain;
109 burg_method(istream_iterator<real>(f), istream_iterator<real>(),
110 mean, maxorder, est.begin(), &sigma2e, &gain, cor.begin(),
111 subtract_mean, false);
112
113 // Close input file
114 f.close();
115
116 // Solve Yule-Walker equations using Zohar's algorithm as consistency check
117 // Given right hand side containing rho_1, ..., rho_p the solution should
118 // be -a_1, ..., -a_p on success so adding to it a_1, ..., a_p gives errors.
119 vector<real> rhs(++cor.begin(), cor.end());
120 zohar_linear_solve(++cor.begin(), --cor.end(), rhs.begin());
121 transform(rhs.begin(), rhs.end(), est.begin(), rhs.begin(), plus<real>());
122 real res = accumulate(rhs.begin(), rhs.end(), real(0), sum_error<real>());
123
124 printf("%22s %22s %22s\n", "Nominal Value", "burg_method", "Percent Diff");
125 printf("%22s %22s %22s\n", "-------------", "-----------", "------------");
126 for (vector<real>::const_iterator i = exact.begin(), j = est.begin();
127 i != exact.end(); ++i, ++j)
128 {
129 printf("%22.14g %22.14g %22.14g\n",
130 (double) *i, (double) *j, (double) pdiff(*i, *j));
131 }
132 printf("\n");
133 printf("%22s %22.14g\n", "mean of data:", (double) mean );
134 printf("%22s %22.14g\n", "\\sigma^2_\\epsilon:", (double) sigma2e );
135 printf("%22s %22.14g\n", "signal gain:", (double) gain );
136 printf("%22s %22.14g\n", "\\sigma^2_x:", (double) (gain*sigma2e));
137 printf("%22s %22.14g\n", "Yule-Walker residual:", (double) res );
138
139 return 0;
140}
