7"""Test the AR Python module using the ar.arsel function."""

15def main():
16 """Test ar.arsel() against known test data."""
17
18 print("Loading required inputs")
19 with open("rhoe.dat") as file:
20 data = [float(line.strip()) for line in file.readlines()]
21 with open("rhoe.coeff") as file:
22 desired = [float(line.strip()) for line in file.readlines()]
23
24 print("Fitting AR model using both positional and keyword arguments")
25 kwargs = {
26 "data": data,
27 "submean": True,
28 "absrho": True,
29 "criterion": "CIC",
30 "minorder": len(desired),
31 "maxorder": len(desired),
32 }
33 result_positional = ar.arsel(*kwargs.values())
34 result_keyword = ar.arsel(**kwargs)
35
36 print("Confirming results match regardless of calling convention")
37 npt.assert_equal(result_positional.N, result_keyword.N)
38 npt.assert_equal(result_positional.AR, result_keyword.AR)
39 npt.assert_equal(result_positional.mu, result_keyword.mu)
40 npt.assert_equal(result_positional.sigma2eps, result_keyword.sigma2eps)
41 npt.assert_equal(result_positional.gain, result_keyword.gain)
42 npt.assert_equal(result_positional.sigma2x, result_keyword.sigma2x)
43 npt.assert_equal(result_positional.autocor, result_keyword.autocor)
44 npt.assert_equal(result_positional.data, result_keyword.data)
45
46 print("Confirming coefficients sufficiently close to expected results")
47 actual = result_positional.AR[0][1:]
48 npt.assert_allclose(actual=actual, desired=desired)

50 print("PASS")
